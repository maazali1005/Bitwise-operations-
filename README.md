# Bitwise-operations
arr1 = np.array([1, 0])

arr2 = np.array([1, 1])

print("Bitwise AND:", np.bitwise_and(arr1, arr2))

print("Bitwise OR:", np.bitwise_or(arr1, arr2)) 

print("Bitwise XOR:", np.bitwise_xor(arr1, arr2))

print("Bitwise NOT:", np.bitwise_not(arr1))
